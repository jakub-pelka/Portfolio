<SectionHeader num="04" title="CODE" id="kod" />

<CodeSnippet
  lang="typescript"
  title="lib/vision/classify.ts"
  annotation="Gemini Vision API analyses the photo and returns a structured object validated against a Zod schema — no manual parsing"
  code={"const schema = z.object({\n  name: z.string(),\n  category: z.enum(['furniture', 'clothing', 'prop', 'decoration']),\n  condition: z.enum(['new', 'good', 'worn', 'damaged']),\n  tags: z.array(z.string()),\n  notes: z.string().optional(),\n});\n\nexport async function classifyProp(imageBase64: string) {\n  const result = await model.generateContent([\n    { inlineData: { data: imageBase64, mimeType: 'image/jpeg' } },\n    CLASSIFY_PROMPT,\n  ]);\n  return schema.parse(JSON.parse(result.response.text()));\n}"}
/>

<CodeSnippet
  lang="typescript"
  title="app/api/props/route.ts"
  annotation="Photo upload → Supabase Storage → Vision classification → database insert — one endpoint handles the entire pipeline"
  code={"export async function POST(req: Request) {\n  const form = await req.formData();\n  const file = form.get('image') as File;\n\n  const { path } = await supabase.storage\n    .from('props')\n    .upload(`${Date.now()}-${file.name}`, file);\n\n  const classification = await classifyProp(\n    await fileToBase64(file)\n  );\n\n  const { data } = await supabase\n    .from('props')\n    .insert({ ...classification, image_path: path })\n    .select()\n    .single();\n\n  return Response.json(data);\n}"}
/>
